var a[10][10],i,j,k;

//利用二维数组打印输出矩阵，并测试矩阵输入
begin
	k := 0;
	for(i:=1; i<6; i:=i+1)
	begin
		for(j:=1; j<6; j:=j+1)
		begin
			k += 1;
			a[i][j] := k;
			print(a[i][j]);
		end;
		print();
	end;
	//read the array.
	for(i:=1; i<=2; i:=i+1)
	begin
		for(j:=1; j<=2; j:=j+1)
		begin
			read(a[i][j]);
			print(a[i][j]);
		end;
	end;
	
end.
